minions项目的数据来源

== 安装依赖 ==

[source, bash]
----
pip install pipenv
#请使用python 3.8
pipenv install
#解析js需要用到splash，暂时未用到此组件
docker run -p 8050:8050 --name -d splash-minions scrapinghub/splash:3.5.0
----

== 运行爬虫 ==

[source, bash]
----
# 查看所有爬虫
scrapy list
# 执行名称未 hero 的爬虫
scrapy crawl hero
----

== 其他命令 ==

[source, bash]
----
# 更新需要更新的依赖
pipenv update --outdated

# 安装某个依赖
pipenv install <package>

# 查看当前爬虫
scrapy list
# 运行指定爬虫
scrapy crawl hero

# 项目部署 安装scrapyd
pipenv install scrapyd
# 启动scrapyd
scrapyd

# 使用scrapy-client将项目部署到scrapyd中
scrapyd-deploy minions_spider -p minions_spider -v v1

# 更换环境
export SCRAPY_PROJECT=test
----
